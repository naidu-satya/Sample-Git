package ProblemsOnBinarySearch;
// reverse of the ceiling of a number
public class FloorOfNumber {
    public static void main(String[] args) {
        int[] arr = {-18, -4, 0, 2, 4, 5, 8, 12, 33, 45, 56 , 89};
        int target = -20;
        int ans = floorOfNumber(arr, target);
        System.out.println(ans);
    }

    //return the greatest no. that is <= to target
    static int floorOfNumber(int[] arr, int target){
        int start = 0;
        int end = arr.length - 1;
        if(target < arr[0])
            return Integer.MAX_VALUE;
        while( start <= end){
            int mid = start + (end - start) / 2;
            if(target > arr[mid]){
                start = mid + 1;
            }
            else if( target < arr[mid]){
                end = mid -1;
            }
            else
                return arr[mid];
        }
        return arr[end];
    }
}
