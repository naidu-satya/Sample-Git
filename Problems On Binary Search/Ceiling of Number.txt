package ProblemsOnBinarySearch;

public class CeilingOfNumber {
    public static void main(String[] args) {
        int[] arr = {-18, -4, 0, 2, 4, 5, 8, 12, 33, 45, 56 , 89};
        int target = 99;
        int ans = ceilingOfNumber(arr, target);
        System.out.println(ans);
    }

    //return the smallest n0. that is >= target
    static int ceilingOfNumber(int[] arr, int target){
        int start = 0;
        int end = arr.length - 1;
        if(target > arr[arr.length - 1])
            return Integer.MIN_VALUE;
        while( start <= end){       // when while loop ends, start = end + 1, so the next big element of the target will be start
            int mid = start + (end - start) / 2;
            if(target > arr[mid]){
                start = mid + 1;
            }
            else if( target < arr[mid]){
                end = mid -1;
            }
            else
                return arr[mid];
        }
        return arr[start];
    }
}
