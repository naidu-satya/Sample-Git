package ProblemsOnBinarySearch;
// Find the first and last position of the given element in an array
public class FirstAndLastIndexOfElement {
    public static void main(String[] args) {

    }
    public int[] searchRange(int[] nums, int target) {
    int[] ans = {-1, -1};
    int start = search(nums, target, true);
    int end = search(nums, target, false);

    ans[0] = start;
    ans[1] = end;
    return ans;
    }

    int search(int[] nums, int target, boolean findfirstIndex){
        int ans = -1;
        int start =0;
        int end = nums.length - 1;
        while(start <= end){
            //find the middle element
            //int mid = (start + end) / 2    //might be possible that it may exceed the integer range
            int mid = start + (end - start) /2;
            if(target < nums[mid])
                end = mid -1;
            else if(target > nums[mid])
                start = mid + 1;
            else {
                ans = mid;
                if (findfirstIndex == true)
                    end = mid - 1;
                else
                    start = mid + 1;
            }
        }
        return ans;
    }
}
