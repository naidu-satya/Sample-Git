*********************Using Type Casting*************************
                    //using type  casting
public class PrintingASCIIValue {
    public static void main(String args[]) {
        //characters whose ASCII value to be found
        char ch1 = 'a';
        char ch2 = 'b';
        ////casting or converting a charter into int type
        int asciiValue1 = (int) ch1;
        int asciiValue2 = (int) ch2;
        System.out.println("ASCII Value of "+ch1+ "is "+asciiValue1);
        System.out.println("ASCII Value of "+ch2+ "is "+asciiValue2);
    }
}
