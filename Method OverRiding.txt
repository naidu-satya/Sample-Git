**********************Real Time Example of Method OverRiding************************
public class MethodOverriding {
    int getRateOfInterest(){
        return 0;
    }
}
class ICICI extends MethodOverriding{
    int getRateOfInterest(){
        return 14;
    }
}
class SBI extends MethodOverriding{
    int getRateOfInterest(){
        return 17;
    }
}
class Axis extends MethodOverriding{
    int getRateOfInterest(){
        return 13;
    }
}
class childClass{
    public static void main(String args[]){
        SBI sbi = new SBI();
        ICICI icici = new ICICI();
        Axis axis = new Axis();
        System.out.println("Rate of interest of ICICI bank is: "+icici.getRateOfInterest());
        System.out.println("Rate of interest of SBI bank is: "+sbi.getRateOfInterest());
        System.out.println("Rate of interest of Axis bank is: "+axis.getRateOfInterest());

    }
}
